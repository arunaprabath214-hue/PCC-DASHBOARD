create table clients ( id uuid default gen_random_uuid() primary key, name text not null, phone text, monthly_bottles int default 0, delivery_day int default 1, price_per_bottle numeric default 35, created_at timestamptz default now() ); create table orders ( id uuid default gen_random_uuid() primary key, client_id uuid references clients(id), order_ref text, qty int, price_per_bottle numeric, paid numeric default 0, order_date date, delivery_date date, stage int default 0, created_at timestamptz default now() );
